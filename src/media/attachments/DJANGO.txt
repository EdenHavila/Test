#############structure d'un projet modulaire et professionnel#####################

src/
│
├── myproject/              # Dossier principal du projet (contenant les settings, urls, wsgi, etc.)
│   ├── __init__.py
│   ├── settings.py         # Paramètres de configuration du projet
│   ├── urls.py             # URLs globales du projet
│   ├── asgi.py
│   ├── wsgi.py
│   └── settings/           # Dossier contenant des fichiers de settings supplémentaires (si nécessaire)
│
├── apps/  		    #les applications du projet
│		    
├── templates/              # Dossier global pour les templates de base
│   ├── base.html           # Template de base (héritage avec 'extends')
│   ├── header.html         # Template de l'en-tête (inclus avec 'include')
│   ├── footer.html         # Template du pied de page (inclus avec 'include')
│
├── static/                 # Dossier pour les fichiers statiques (CSS, JS, images)
│
├── media/                  # Dossier pour les fichiers médias (uploads d'images, etc.)
│
└── manage.py               # Script pour gérer le projet Django (migrations, serveur, etc.)
